```typescript
// errors.ts
import type { ContractAwarenessIssue, ContractAwarenessErrorCode } from "./types.js";

export function createIssue(params: {
  code: ContractAwarenessErrorCode;
  category: "contract" | "cache" | "schema" | "fingerprint";
  severity: "ERROR" | "WARN";
  message: string;
  details?: Record<string, unknown>;
}): ContractAwarenessIssue {
  return {
    code: params.code,
    category: params.category,
    severity: params.severity,
    message: params.message,
    ...(params.details && { details: params.details })
  };
}

export function getSeverity(
  code: ContractAwarenessErrorCode,
  mode: "COMPLIANCE" | "BEST_EFFORT"
): "ERROR" | "WARN" {
  // Cache issues are ALWAYS warnings
  if (code.startsWith("CA_CACHE_")) return "WARN";

  // Catastrophic issues are always ERROR (even in BEST_EFFORT)
  if (code === "CA_SCHEMA_INVALID" || code === "CA_JSON_PARSE_ERROR") {
    return "ERROR";
  }

  // Other contract/fingerprint issues depend on mode
  if (mode === "COMPLIANCE") return "ERROR";
  return "WARN";
}
```

```typescript
// loader.ts
import { existsSync, readFileSync, mkdirSync } from "fs";
import { join } from "path";
import type { LoadContractBundleParams, LoadContractBundleResult, ContractAwarenessIssue } from "./types.js";
import { createIssue } from "./errors.js";

const REQUIRED_FILES = [
  "automation-contract.json",
  "framework-pattern.json",
  "selector-strategy.json",
  "assertion-style.json",
  "page-key-policy.json",
  "repo-topology.json",
  "wrapper-discovery.json",
  "automation-contract.hash"
] as const;

function resolveContractDir(repoRoot: string): {
  dir: string | null;
  isLegacy: boolean;
  issues: ContractAwarenessIssue[];
} {
  const canonical = join(repoRoot, ".mcp-contract");
  const legacy = join(repoRoot, ".mindtrace/contracts");

  // 1. Try canonical first
  if (existsSync(canonical)) {
    return { dir: canonical, isLegacy: false, issues: [] };
  }

  // 2. Fallback to legacy
  if (existsSync(legacy)) {
    return {
      dir: legacy,
      isLegacy: true,
      issues: [
        createIssue({
          code: "CA_LEGACY_PATH",
          category: "contract",
          severity: "WARN",
          message: "Using legacy contract directory .mindtrace/contracts/ (migrate to .mcp-contract/)"
        })
      ]
    };
  }

  // 3. Neither exists
  return {
    dir: null,
    isLegacy: false,
    issues: [
      createIssue({
        code: "CA_CONTRACT_DIR_MISSING",
        category: "contract",
        severity: "ERROR",
        message: "Contract directory not found (.mcp-contract/ or .mindtrace/contracts/)"
      })
    ]
  };
}

export async function loadContractBundle(
  params: LoadContractBundleParams
): Promise<LoadContractBundleResult> {
  const { repoRoot, mode } = params;
  const issues: ContractAwarenessIssue[] = [];

  // Resolve contract directory
  const { dir: contractDir, isLegacy, issues: dirIssues } = resolveContractDir(repoRoot);
  issues.push(...dirIssues);

  if (!contractDir) {
    return {
      ok: false,
      error: "Contract directory not found",
      issues
    };
  }

  // Load required files
  const files: Record<string, any> = {};
  for (const file of REQUIRED_FILES) {
    const filePath = join(contractDir, file);
    if (!existsSync(filePath)) {
      const severity = mode === "COMPLIANCE" ? "ERROR" : "WARN";
      issues.push(
        createIssue({
          code: "CA_MISSING_FILE",
          category: "contract",
          severity,
          message: `Required file not found: ${file}`
        })
      );
      continue;
    }

    try {
      const content = readFileSync(filePath, "utf-8");
      if (file.endsWith(".json")) {
        files[file] = JSON.parse(content);
      } else if (file.endsWith(".hash")) {
        files[file] = content.trim();
      }
    } catch (error) {
      issues.push(
        createIssue({
          code: "CA_JSON_PARSE_ERROR",
          category: "schema",
          severity: "ERROR",
          message: `Failed to parse ${file}: ${error instanceof Error ? error.message : String(error)}`
        })
      );
    }
  }

  // Read contract hash
  const contractHash = files["automation-contract.hash"] || "";

  return {
    ok: true,
    contractDir,
    isLegacy,
    contractHash,
    cacheDir: null,  // Will be resolved by bindCache
    cacheIsLegacy: false,
    repoRoot,
    files,
    issues
  };
}
```

```typescript
// validator.ts
import { readFileSync, existsSync } from "fs";
import { join } from "path";
import Ajv from "ajv";
import addFormats from "ajv-formats";
import { computeContractFingerprint } from "../../../repo-intelligence-mcp/src/tools/fingerprintContract.js";
import type { ValidateContractBundleParams, ContractValidationResult, ContractAwarenessIssue } from "./types.js";
import { createIssue, getSeverity } from "./errors.js";

const SCHEMA_MAP: Record<string, string> = {
  "automation-contract.json": "automation-contract.schema.json",
  "framework-pattern.json": "framework-pattern.schema.json",
  "selector-strategy.json": "selector-strategy.schema.json",
  "assertion-style.json": "assertion-style.schema.json",
  "page-key-policy.json": "page-key-policy.schema.json"
};

export async function validateContractBundle(
  params: ValidateContractBundleParams
): Promise<ContractValidationResult> {
  const { loaded, mode } = params;
  const issues: ContractAwarenessIssue[] = [...loaded.issues];

  // 1. Schema validation
  const schemaIssues = validateSchemas(loaded.files, mode);
  issues.push(...schemaIssues);

  // 2. Fingerprint verification
  const fingerprintResult = verifyFingerprint(loaded.contractDir, mode);
  issues.push(...fingerprintResult.issues);

  const valid = !issues.some(i => i.severity === "ERROR");

  return { valid, issues };
}

function validateSchemas(
  files: Record<string, any>,
  mode: "COMPLIANCE" | "BEST_EFFORT"
): ContractAwarenessIssue[] {
  const issues: ContractAwarenessIssue[] = [];
  const ajv = new Ajv({ allErrors: true, strict: false });
  addFormats(ajv);

  // Load schemas
  const schemasDir = join(__dirname, "schemas");

  for (const [file, schemaFile] of Object.entries(SCHEMA_MAP)) {
    const data = files[file];
    if (!data) continue;

    const schemaPath = join(schemasDir, schemaFile);
    if (!existsSync(schemaPath)) continue;

    try {
      const schema = JSON.parse(readFileSync(schemaPath, "utf-8"));
      const validate = ajv.compile(schema);
      const valid = validate(data);

      if (!valid) {
        issues.push(
          createIssue({
            code: "CA_SCHEMA_INVALID",
            category: "schema",
            severity: "ERROR",  // Always ERROR
            message: `Schema validation failed for ${file}`,
            details: { errors: validate.errors, file }
          })
        );
      }
    } catch (error) {
      issues.push(
        createIssue({
          code: "CA_SCHEMA_INVALID",
          category: "schema",
          severity: "ERROR",
          message: `Failed to validate ${file}: ${error instanceof Error ? error.message : String(error)}`
        })
      );
    }
  }

  return issues;
}

export function verifyFingerprint(
  contractDir: string,
  mode: "COMPLIANCE" | "BEST_EFFORT"
): { match: boolean; issues: ContractAwarenessIssue[] } {
  const issues: ContractAwarenessIssue[] = [];
  const hashPath = join(contractDir, "automation-contract.hash");

  if (!existsSync(hashPath)) {
    const severity = getSeverity("CA_HASH_MISSING", mode);
    issues.push(
      createIssue({
        code: "CA_HASH_MISSING",
        category: "fingerprint",
        severity,
        message: "automation-contract.hash not found"
      })
    );
    return { match: false, issues };
  }

  const storedHash = readFileSync(hashPath, "utf-8").trim();
  const fingerprintMode = mode === "COMPLIANCE" ? "strict" : "best_effort";
  const computed = computeContractFingerprint(contractDir, fingerprintMode);

  if (!computed.ok) {
    issues.push(
      createIssue({
        code: "CA_HASH_COMPUTE_FAILED",
        category: "fingerprint",
        severity: getSeverity("CA_HASH_COMPUTE_FAILED", mode),
        message: computed.error
      })
    );
    return { match: false, issues };
  }

  const match = computed.fingerprint === storedHash;

  if (!match) {
    const severity = getSeverity("CA_HASH_MISMATCH", mode);
    issues.push(
      createIssue({
        code: "CA_HASH_MISMATCH",
        category: "fingerprint",
        severity,
        message: `Fingerprint mismatch. Expected: ${storedHash}, Got: ${computed.fingerprint}`,
        details: { stored: storedHash, computed: computed.fingerprint }
      })
    );
  }

  return { match, issues };
}
```

```typescript
// cacheBinding.ts
import { existsSync, readFileSync } from "fs";
import { join } from "path";
import type { BindCacheParams, CacheBindingResult, ContractAwarenessIssue } from "./types.js";
import { createIssue } from "./errors.js";

function resolveCacheDir(repoRoot: string): {
  dir: string | null;
  isLegacy: boolean;
  metaPath: string | null;
  issues: ContractAwarenessIssue[];
} {
  const canonicalV1 = join(repoRoot, ".mcp-cache/v1");
  const legacyRoot = join(repoRoot, ".mcp-cache");

  // 1. Try canonical v1/ structure
  const canonicalMetaPath = join(canonicalV1, "meta.json");
  if (existsSync(canonicalMetaPath)) {
    return { dir: canonicalV1, isLegacy: false, metaPath: canonicalMetaPath, issues: [] };
  }

  // 2. Fallback to legacy index.json
  const legacyIndexPath = join(legacyRoot, "index.json");
  if (existsSync(legacyIndexPath)) {
    return {
      dir: legacyRoot,
      isLegacy: true,
      metaPath: legacyIndexPath,
      issues: [
        createIssue({
          code: "CA_CACHE_LEGACY_PATH",
          category: "cache",
          severity: "WARN",
          message: "Using legacy cache structure (index.json)"
        })
      ]
    };
  }

  // 3. Cache not found (non-fatal)
  return {
    dir: null,
    isLegacy: false,
    metaPath: null,
    issues: [
      createIssue({
        code: "CA_CACHE_META_MISSING",
        category: "cache",
        severity: "WARN",
        message: "Page cache not found (cache is optional)"
      })
    ]
  };
}

export function bindCache(params: BindCacheParams): CacheBindingResult {
  const { repoRoot, contractHash } = params;
  const issues: ContractAwarenessIssue[] = [];

  // Resolve cache directory
  const { dir: cacheDir, isLegacy, metaPath, issues: dirIssues } = resolveCacheDir(repoRoot);
  issues.push(...dirIssues);

  if (!cacheDir || !metaPath) {
    return {
      valid: false,
      cacheHash: null,
      contractHash,
      pagesCount: 0,
      cacheDir: null,
      isLegacy: false,
      issues
    };
  }

  // Read cache metadata
  try {
    const content = readFileSync(metaPath, "utf-8");
    const meta = JSON.parse(content);

    let cacheHash: string | null = null;
    let pagesCount = 0;

    if (isLegacy) {
      // Legacy: index.json structure
      cacheHash = meta.contractSha256 || null;
      pagesCount = meta.pages?.length || 0;
    } else {
      // Canonical: v1/meta.json structure
      cacheHash = meta.contract_hash || null;
      pagesCount = meta.pages_count || 0;
    }

    const match = cacheHash === contractHash;

    if (!match) {
      issues.push(
        createIssue({
          code: "CA_CACHE_HASH_MISMATCH",
          category: "cache",
          severity: "WARN",  // Always WARN
          message: `Cache hash mismatch. Expected: ${contractHash}, Got: ${cacheHash || "null"}. Cache will be ignored.`,
          details: { contractHash, cacheHash }
        })
      );
    }

    return {
      valid: match,
      cacheHash,
      contractHash,
      pagesCount: match ? pagesCount : 0,  // Zero if invalid
      cacheDir: match ? cacheDir : null,
      isLegacy,
      issues
    };
  } catch (error) {
    issues.push(
      createIssue({
        code: "CA_CACHE_READ_ERROR",
        category: "cache",
        severity: "WARN",
        message: `Failed to read cache metadata: ${error instanceof Error ? error.message : String(error)}`
      })
    );

    return {
      valid: false,
      cacheHash: null,
      contractHash,
      pagesCount: 0,
      cacheDir: null,
      isLegacy,
      issues
    };
  }
}
```

```typescript
// canonicalStringify.ts
export function canonicalStringify(obj: any): string {
  return JSON.stringify(obj, (key, val) => {
    if (val && typeof val === "object" && !Array.isArray(val)) {
      // Sort object keys recursively
      return Object.keys(val)
        .sort()
        .reduce((sorted, k) => {
          sorted[k] = val[k];
          return sorted;
        }, {} as any);
    }
    return val;
  }, 2) + "\n";  // Trailing newline
}
```

```typescript
// writer.ts
import { writeFileSync, renameSync, mkdirSync, existsSync } from "fs";
import { join, dirname, basename } from "path";
import { randomUUID } from "crypto";
import type { WriteRuntimeContextParams } from "./types.js";
import { canonicalStringify } from "./canonicalStringify.js";

function determineArtifactPath(artifactsDir: string): string {
  // Check if legacy path exists (Phase 2.2 backward compat)
  const legacyPath = join(artifactsDir, "runtime-contract-context.json");
  if (existsSync(legacyPath)) {
    return legacyPath;
  }

  // Use phase2/ directory (new convention)
  const phase2Path = join(artifactsDir, "phase2", "runtime-contract-context.json");
  mkdirSync(dirname(phase2Path), { recursive: true });
  return phase2Path;
}

export function writeRuntimeContractContext(
  params: WriteRuntimeContextParams
): string {
  const { artifactsDir, runId, loaded, validation, cache, strategy } = params;

  const output = {
    schema_version: "1.0.0",
    runId: runId || basename(dirname(artifactsDir)),
    contract: {
      hash: loaded.contractHash,
      schema_version: strategy.contract.schemaVersion,
      framework: strategy.framework,
      directory: strategy.contract.directory,
      isLegacy: strategy.contract.isLegacy
    },
    cache: {
      valid: cache.valid,
      hash_match: cache.valid,
      pages_loaded: cache.pagesCount,
      directory: strategy.cache.directory,
      isLegacy: strategy.cache.isLegacy
    },
    strategy,
    issues: validation.issues.map(i => ({
      code: i.code,
      category: i.category,
      severity: i.severity,
      message: i.message
    }))
  };

  const outputPath = determineArtifactPath(artifactsDir);
  const tempPath = outputPath + ".tmp." + randomUUID();

  // Atomic write: temp → rename
  writeFileSync(tempPath, canonicalStringify(output), "utf-8");
  renameSync(tempPath, outputPath);

  return outputPath;
}
```

```typescript
export function applyRuntimeContractContextEnv(args: {
  artifactsDir: string;
}): RuntimeContractContext {
  const mode = process.env.MINDTRACE_MODE === "strict" ? "COMPLIANCE" : "BEST_EFFORT";
  const repoRoot = findRepoRoot(args.artifactsDir);

  // Phase 2.0: Full contract loading + validation
  const loaded = await loadContractBundle({ repoRoot, mode });

  if (!loaded.ok) {
    return {
      ok: false,
      exitCode: mode === "COMPLIANCE" ? 3 : 0,
      source: "validation",
      repoRoot: null,
      contractDir: null,
      cacheDir: null,
      warnings: [],
      files: {},
      summary: {},
      notes: [loaded.error]
    };
  }

  const validation = await validateContractBundle({ loaded, mode });

  // Check for ERROR issues
  const hasErrors = validation.issues.some(i => i.severity === "ERROR");

  // COMPLIANCE gate: Write validation artifact and return exit signal
  if (mode === "COMPLIANCE" && hasErrors) {
    const validationPath = join(args.artifactsDir, "phase0", "validation.json");
    mkdirSync(dirname(validationPath), { recursive: true });
    writeFileSync(validationPath, canonicalStringify({
      valid: false,
      issues: validation.issues
    }), "utf-8");

    return {
      ok: false,
      exitCode: 3,
      source: "validation",
      repoRoot: loaded.repoRoot,
      contractDir: null,
      cacheDir: null,
      warnings: [],
      files: {},
      summary: {},
      notes: [`Contract validation failed (${validation.issues.length} issues)`]
    };
  }

  // Bind cache (warnings only)
  const cache = bindCache({ repoRoot, contractHash: loaded.contractHash });

  // Build strategy context
  const strategy = buildStrategyContext({ loaded, cache });

  // Write runtime context artifact
  writeRuntimeContractContext({
    artifactsDir: args.artifactsDir,
    loaded,
    validation,
    cache,
    strategy
  });

  // Set environment variables (Phase 2.2 backward compat)
  process.env.MINDTRACE_CONTRACT_DIR = loaded.contractDir;
  if (cache.valid) {
    process.env.MINDTRACE_PAGE_CACHE_DIR = loaded.cacheDir || "";
  }
  process.env.MINDTRACE_AUTOMATION_CONTRACT_CONTEXT_PATH =
    join(args.artifactsDir, "runtime-contract-context.json");

  // Return legacy format for backward compatibility
  return {
    ok: true,
    source: "artifacts",
    repoRoot: loaded.repoRoot,
    contractDir: loaded.contractDir,
    cacheDir: cache.valid ? loaded.cacheDir : null,
    warnings: validation.issues
      .filter(i => i.severity === "WARN")
      .map(i => `${i.code}: ${i.message}`),
    files: loaded.files,
    summary: strategy,
    notes: []
  };
}
```

```typescript
    artifactsDir,

    strategy
```
